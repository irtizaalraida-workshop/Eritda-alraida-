    // initialize year
    document.getElementById('year').textContent = new Date().getFullYear();

    // Language toggle
    const setLang = (lang) => {
      const root = document.documentElement;
      const isAR = lang === 'ar';
      // change page direction
      if(isAR){
        root.classList.add('arabic');
        root.dir = 'rtl';
      } else {
        root.classList.remove('arabic');
        root.dir = 'ltr';
      }

      // toggle visibility of text blocks
      const enBlocks = document.querySelectorAll('[id$="-en"], #title-en, #lead-en, #services-en');
      const arBlocks = document.querySelectorAll('[id$="-ar"], #title-ar, #lead-ar, #services-ar');

      // A safer toggling: manually list known ids
      const idsEn = ['title-en','lead-en','services-en','cta-en','about-title-en','about-text-en','services-title-en','s1-en','s1-desc-en','s2-en','s2-desc-en','s3-en','s3-desc-en','contact-title-en','lbl-name-en','lbl-email-en','lbl-message-en','note-en','visit-en','addr-en'];
      const idsAr = ['title-ar','lead-ar','services-ar','cta-ar','about-title-ar','about-text-ar','services-title-ar','s1-ar','s1-desc-ar','s2-ar','s2-desc-ar','s3-ar','s3-desc-ar','contact-title-ar','lbl-name-ar','lbl-email-ar','lbl-message-ar','note-ar','visit-ar','addr-ar'];

      idsEn.forEach(id => { const el = document.getElementById(id); if(el) el.style.display = isAR ? 'none' : 'block'; });
      idsAr.forEach(id => { const el = document.getElementById(id); if(el) el.style.display = isAR ? 'block' : 'none'; });

      // adjust hero layout for arabic (swap columns)
      const hero = document.querySelector('.hero');
      if(hero){
        hero.style.gridTemplateColumns = isAR ? '420px 1fr' : '1fr 420px';
      }
    };

    // default language: show both but english by default
    setLang('en');

    // Contact form submit handler: simple mailto fallback + feedback
    function submitForm(e){
      e.preventDefault();
      const name = document.getElementById('name').value.trim();
      const email = document.getElementById('email').value.trim();
      const message = document.getElementById('message').value.trim();
      const subject = encodeURIComponent('Website contact from ' + name);
      const body = encodeURIComponent('Name: ' + name + '\\nEmail: ' + email + '\\n\\n' + message);
      // fallback: open mailto
      window.location.href = `mailto:irtizaalraida@gmail.com?subject=${subject}&body=${body}`;
    }
